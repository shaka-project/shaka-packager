235 Status MediaHandlerTestBase::SetUpAndInitializeGraph(
236  std::shared_ptr<MediaHandler> handler,
237  size_t input_count,
238  size_t output_count) {
239  DCHECK(handler);
240  DCHECK_EQ(nullptr, handler_);
241  DCHECK(inputs_.empty());
242  DCHECK(outputs_.empty());
243 
244  handler_ = std::move(handler);
245 
246  Status status;
247 
248  // Add and connect all the requested inputs.
249  for (size_t i = 0; i < input_count; i++) {
250  inputs_.emplace_back(new FakeInputMediaHandler);
251  }
252 
253  for (auto& input : inputs_) {
254  status.Update(input->AddHandler(handler_));
255  }
256 
257  if (!status.ok()) {
258  return status;
259  }
260 
261  // Add and connect all the requested outputs.
262  for (size_t i = 0; i < output_count; i++) {
263  outputs_.emplace_back(new testing::NiceMock<MockOutputMediaHandler>);
264  }
265 
266  for (auto& output : outputs_) {
267  status.Update(handler_->AddHandler(output));
268  }
269 
270  if (!status.ok()) {
271  return status;
272  }
273 
274  // Initialize the graph.
275  for (auto& input : inputs_) {
276  status.Update(input->Initialize());
277  }
278 
279  // In the case that there are no inputs, the start of the graph
280  // is at |handler_| so it needs to be initialized or else the graph
281  // won't be initialized.
282  if (inputs_.empty()) {
283  status.Update(handler_->Initialize());
284  }
285 
286  return status;
287 }

304  size_t num_inputs,

307  // Input handler is not really used anywhere but just to satisfy one input
