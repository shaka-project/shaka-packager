146 std::shared_ptr<StreamInfo> MediaHandlerTestBase::GetMockStreamInfo(
147  Codec codec,
148  uint32_t time_scale) {
149  if (codec >= kCodecAudio && codec < kCodecAudioMaxPlusOne) {
150  return std::shared_ptr<StreamInfo>(new AudioStreamInfo(
151  kTrackId, time_scale, kDuration, codec, kCodecString, kCodecConfig,
152  sizeof(kCodecConfig), kSampleBits, kNumChannels, kSamplingFrequency,
153  kSeekPrerollNs, kCodecDelayNs, kMaxBitrate, kAvgBitrate, kLanguage,
154  !kEncrypted));
155  } else if (codec >= kCodecVideo && codec < kCodecVideoMaxPlusOne) {
156  return std::shared_ptr<StreamInfo>(new VideoStreamInfo(
157  kTrackId, time_scale, kDuration, codec, H26xStreamFormat::kUnSpecified,
158  kCodecString, kCodecConfig, sizeof(kCodecConfig), kWidth, kHeight,
159  kPixelWidth, kPixelHeight, kTrickPlayFactor, kNaluLengthSize, kLanguage,
160  !kEncrypted));
161  }
162  return nullptr;
163 }
