29 bool PesPacketGenerator::Initialize(const StreamInfo& stream_info) {
30  STLDeleteElements(&pes_packets_);
31  stream_type_ = stream_info.stream_type();
32 
33  if (stream_type_ == kStreamVideo) {
34  const VideoStreamInfo& video_stream_info =
35  static_cast<const VideoStreamInfo&>(stream_info);
36  if (video_stream_info.codec() != VideoCodec::kCodecH264) {
37  NOTIMPLEMENTED() << "Video codec " << video_stream_info.codec()
38  << " is not supported.";
39  return false;
40  }
41  timescale_scale_ = 90000.0 / video_stream_info.time_scale();
42  converter_.reset(new NalUnitToByteStreamConverter());
43  return converter_->Initialize(video_stream_info.extra_data().data(),
44  video_stream_info.extra_data().size(),
45  !kEscapeData);
46  } else if (stream_type_ == kStreamAudio) {
47  const AudioStreamInfo& audio_stream_info =
48  static_cast<const AudioStreamInfo&>(stream_info);
49  if (audio_stream_info.codec() != AudioCodec::kCodecAAC) {
50  NOTIMPLEMENTED() << "Audio codec " << audio_stream_info.codec()
51  << " is not supported yet.";
52  return false;
53  }
54  timescale_scale_ = 90000.0 / audio_stream_info.time_scale();
55  adts_converter_.reset(new mp4::AACAudioSpecificConfig());
56  return adts_converter_->Parse(audio_stream_info.extra_data());
57  }
58 
59  NOTIMPLEMENTED() << "Stream type: " << stream_type_ << " not implemented.";
60  return false;
61 }

63 bool PesPacketGenerator::PushSample(scoped_refptr<MediaSample> sample) {
64  if (!current_processing_pes_)
65  current_processing_pes_.reset(new PesPacket());
66 
67  current_processing_pes_->set_pts(timescale_scale_ * sample->pts());
68  current_processing_pes_->set_dts(timescale_scale_ * sample->dts());
69  if (stream_type_ == kStreamVideo) {
70  DCHECK(converter_);
71  if (!converter_->ConvertUnitToByteStream(
72  sample->data(), sample->data_size(), sample->is_key_frame(),
73  current_processing_pes_->mutable_data())) {
74  LOG(ERROR) << "Failed to convert sample to byte stream.";
75  return false;
76  }
77  current_processing_pes_->set_stream_id(kVideoStreamId);
78  pes_packets_.push_back(current_processing_pes_.release());
79  return true;
80  }
81  DCHECK_EQ(stream_type_, kStreamAudio);
82  DCHECK(adts_converter_);
83 
84  std::vector<uint8_t> aac_frame(sample->data(),
85  sample->data() + sample->data_size());
86  if (!adts_converter_->ConvertToADTS(&aac_frame))
87  return false;
88 
89  // TODO(rkuriowa): Put multiple samples in the PES packet to reduce # of PES
90  // packets.
91  current_processing_pes_->mutable_data()->swap(aac_frame);
92  current_processing_pes_->set_stream_id(kAudioStreamId);
93  pes_packets_.push_back(current_processing_pes_.release());
94  return true;
95 }
