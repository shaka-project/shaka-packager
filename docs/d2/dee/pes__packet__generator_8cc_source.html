34 // |target_data| is input as well as output. On success |target_data| contains
35 // the encrypted sample. The input data should be Nal unit byte stream.
36 // This function constructs encrypted sample in |encrypted_sample_data| then
37 // swap with target_data on success.
38 bool EncryptH264Sample(AesCryptor* encryptor,
39  std::vector<uint8_t>* target_data) {
40  BufferWriter encrypted_sample_data(target_data->size() * 1.5);
41 
42  const int kLeadingClearBytesSize = 32;
43  // Any Nalu smaller than 48 bytes shall not be encrypted.
44  const uint64_t kSmallNalUnitSize = 48;
45 
46  NaluReader nalu_reader(Nalu::kH264, 0, target_data->data(),
47  target_data->size());
48  NaluReader::Result result;
49  Nalu nalu;
50  while ((result = nalu_reader.Advance(&nalu)) == NaluReader::Result::kOk) {
51  encrypted_sample_data.AppendInt(static_cast<uint32_t>(0x00000001));
52  const uint64_t nalu_total_size = nalu.header_size() + nalu.payload_size();
53  if (nalu.type() != Nalu::H264NaluType::H264_NonIDRSlice &&
54  nalu.type() != Nalu::H264NaluType::H264_IDRSlice) {
55  VLOG(3) << "Found Nalu type: " << nalu.type() << " skipping encryption.";
56  encrypted_sample_data.AppendArray(nalu.data(), nalu_total_size);
57  continue;
58  }
59 
60  if (nalu_total_size <= kSmallNalUnitSize) {
61  encrypted_sample_data.AppendArray(nalu.data(), nalu_total_size);
62  continue;
63  }
64 
65  const uint8_t* current = nalu.data() + kLeadingClearBytesSize;
66  const uint64_t bytes_remaining = nalu_total_size - kLeadingClearBytesSize;
67 
68  if (!encryptor->Crypt(current, bytes_remaining,
69  const_cast<uint8_t*>(current))) {
70  return false;
71  }
72  EscapeNalByteSequence(nalu.data(), nalu_total_size, &encrypted_sample_data);
73  }
74 
75  encrypted_sample_data.SwapBuffer(target_data);
76  return true;
77 }

81  const int kUnencryptedLeaderSize = 16;

93 bool PesPacketGenerator::Initialize(const StreamInfo& stream_info) {
94  pes_packets_.clear();
95  stream_type_ = stream_info.stream_type();
96 
97  if (stream_type_ == kStreamVideo) {
98  const VideoStreamInfo& video_stream_info =
99  static_cast<const VideoStreamInfo&>(stream_info);
100  if (video_stream_info.codec() != Codec::kCodecH264) {
101  NOTIMPLEMENTED() << "Video codec " << video_stream_info.codec()
102  << " is not supported.";
103  return false;
104  }
105  timescale_scale_ = kTsTimescale / video_stream_info.time_scale();
106  converter_.reset(new NalUnitToByteStreamConverter());
107  return converter_->Initialize(video_stream_info.codec_config().data(),
108  video_stream_info.codec_config().size(),
109  !kEscapeData);
110  } else if (stream_type_ == kStreamAudio) {
111  const AudioStreamInfo& audio_stream_info =
112  static_cast<const AudioStreamInfo&>(stream_info);
113  if (audio_stream_info.codec() != Codec::kCodecAAC) {
114  NOTIMPLEMENTED() << "Audio codec " << audio_stream_info.codec()
115  << " is not supported yet.";
116  return false;
117  }
118  timescale_scale_ = kTsTimescale / audio_stream_info.time_scale();
119  adts_converter_.reset(new AACAudioSpecificConfig());
120  return adts_converter_->Parse(audio_stream_info.codec_config());
121  }
122 
123  NOTIMPLEMENTED() << "Stream type: " << stream_type_ << " not implemented.";
124  return false;
125 }

127 bool PesPacketGenerator::PushSample(std::shared_ptr<MediaSample> sample) {
128  if (!current_processing_pes_)
129  current_processing_pes_.reset(new PesPacket());
130 
131  current_processing_pes_->set_pts(timescale_scale_ * sample->pts());
132  current_processing_pes_->set_dts(timescale_scale_ * sample->dts());
133  if (stream_type_ == kStreamVideo) {
134  DCHECK(converter_);
135  std::vector<uint8_t> byte_stream;
136  if (!converter_->ConvertUnitToByteStream(
137  sample->data(), sample->data_size(), sample->is_key_frame(),
138  &byte_stream)) {
139  LOG(ERROR) << "Failed to convert sample to byte stream.";
140  return false;
141  }
142 
143  if (encryptor_) {
144  if (!EncryptH264Sample(encryptor_.get(), &byte_stream)) {
145  LOG(ERROR) << "Failed to encrypt byte stream.";
146  return false;
147  }
148  }
149  current_processing_pes_->mutable_data()->swap(byte_stream);
150  current_processing_pes_->set_stream_id(kVideoStreamId);
151  pes_packets_.push_back(std::move(current_processing_pes_));
152  return true;
153  }
154  DCHECK_EQ(stream_type_, kStreamAudio);
155  DCHECK(adts_converter_);
156 
157  std::vector<uint8_t> aac_frame(sample->data(),
158  sample->data() + sample->data_size());
159 
160  if (encryptor_) {
161  if (!EncryptAacSample(encryptor_.get(), &aac_frame)) {
162  LOG(ERROR) << "Failed to encrypt ADTS AAC.";
163  return false;
164  }
165  }
166 
167  // TODO(rkuroiwa): ConvertToADTS() makes another copy of aac_frame internally.
168  // Optimize copying in this function, possibly by adding a method on
169  // AACAudioSpecificConfig that takes {pointer, length} pair and returns a
170  // vector that has the ADTS header.
171  if (!adts_converter_->ConvertToADTS(&aac_frame))
172  return false;
173 
174  // TODO(rkuriowa): Put multiple samples in the PES packet to reduce # of PES
175  // packets.
176  current_processing_pes_->mutable_data()->swap(aac_frame);
177  current_processing_pes_->set_stream_id(kAudioStreamId);
178  pes_packets_.push_back(std::move(current_processing_pes_));
179  return true;
180 }

182 bool PesPacketGenerator::SetEncryptionKey(
183  std::unique_ptr<EncryptionKey> encryption_key) {
184  if (stream_type_ == kStreamVideo) {
185  std::unique_ptr<AesCbcEncryptor> cbc(
186  new AesCbcEncryptor(CbcPaddingScheme::kNoPadding));
187 
188  const uint8_t kEncryptedBlocks = 1;
189  const uint8_t kClearBlocks = 9;
190  encryptor_.reset(new AesPatternCryptor(
191  kEncryptedBlocks, kClearBlocks,
192  AesPatternCryptor::kSkipIfCryptByteBlockRemaining,
193  AesCryptor::ConstantIvFlag::kUseConstantIv, std::move(cbc)));
194  } else if (stream_type_ == kStreamAudio) {
195  encryptor_.reset(
196  new AesCbcEncryptor(CbcPaddingScheme::kNoPadding,
197  AesCryptor::ConstantIvFlag::kUseConstantIv));
198  } else {
199  LOG(ERROR) << "Cannot encrypt stream type: " << stream_type_;
200  return false;
201  }
202 
203  return encryptor_->InitializeWithIv(encryption_key->key, encryption_key->iv);
