69  " - bandwidth (bw): Optional value which contains a user-specified\n"
70  " content bit rate for the stream, in bits/sec. If specified, this\n"
71  " value is propagated to (HLS) EXT-X-STREAM-INF:BANDWIDTH or (DASH)\n"
72  " Representation@bandwidth and the $Bandwidth$ template parameter for\n"
73  " segment names. If not specified, the bandwidth value is estimated\n"
74  " from content bitrate. Note that it only affects the generated\n"

288 int PackagerMain(int argc, char** argv) {
289  // Needed to enable VLOG/DVLOG through --vmodule or --v.
290  base::CommandLine::Init(argc, argv);
291 
292  // Set up logging.
293  logging::LoggingSettings log_settings;
294  log_settings.logging_dest = logging::LOG_TO_SYSTEM_DEBUG_LOG;
295  CHECK(logging::InitLogging(log_settings));
296 
297  google::SetVersionString(shaka::Packager::GetLibraryVersion());
298  google::SetUsageMessage(base::StringPrintf(kUsage, argv[0]));
299  google::ParseCommandLineFlags(&argc, &argv, true);
300  if (argc < 2) {
301  google::ShowUsageWithFlags("Usage");
302  return kSuccess;
303  }
304 
305  if (!ValidateWidevineCryptoFlags() || !ValidateFixedCryptoFlags() ||
306  !ValidatePRCryptoFlags()) {
307  return kArgumentValidationFailed;
308  }
309 
310  base::Optional<PackagingParams> packaging_params = GetPackagingParams();
311  if (!packaging_params)
312  return kArgumentValidationFailed;
313 
314  std::vector<StreamDescriptor> stream_descriptors;
315  for (int i = 1; i < argc; ++i) {
316  base::Optional<StreamDescriptor> stream_descriptor =
317  ParseStreamDescriptor(argv[i]);
318  if (!stream_descriptor)
319  return kArgumentValidationFailed;
320  stream_descriptors.push_back(stream_descriptor.value());
321  }
322  Packager packager;
323  Status status =
324  packager.Initialize(packaging_params.value(), stream_descriptors);
325  if (!status.ok()) {
326  LOG(ERROR) << "Failed to initialize packager: " << status.ToString();
327  return kArgumentValidationFailed;
328  }
329  status = packager.Run();
330  if (!status.ok()) {
331  LOG(ERROR) << "Packaging Error: " << status.ToString();
332  return kPackagingFailed;
333  }
334  printf("Packaging completed successfully.\n");
335  return kSuccess;
336 }
