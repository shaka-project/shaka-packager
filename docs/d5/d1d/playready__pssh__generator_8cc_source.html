41 // Converts the key_id's endianness.
42 std::vector<uint8_t> ConvertGuidEndianness(const std::vector<uint8_t>& input) {
43  std::vector<uint8_t> output = input;
44  if (output.size() > 7) { // Defensive check.
45  output[0] = input[3];
46  output[1] = input[2];
47  output[2] = input[1];
48  output[3] = input[0];
49  output[4] = input[5];
50  output[5] = input[4];
51  output[6] = input[7];
52  output[7] = input[6];
53  // 8-15 are an array of bytes with no endianness.
54  }
55  return output;
56 }

58 // Generates the data section of a PlayReady PSSH.
59 // PlayReady PSSH Data is a PlayReady Header Object.
60 // Format is outlined in the following document.
61 // http://download.microsoft.com/download/2/3/8/238F67D9-1B8B-48D3-AB83-9C00112268B2/PlayReady%20Header%20Object%202015-08-13-FINAL-CL.PDF
62 Status GeneratePlayReadyPsshData(const std::vector<uint8_t>& key_id,
63  const std::vector<uint8_t>& key,
64  std::vector<uint8_t>* output) {
65  CHECK(output);
66  std::vector<uint8_t> key_id_converted = ConvertGuidEndianness(key_id);
67  std::vector<uint8_t> encrypted_key_id(key_id_converted.size());
68  std::unique_ptr<AES_KEY> aes_key(new AES_KEY);
69  CHECK_EQ(AES_set_encrypt_key(key.data(), key.size() * 8, aes_key.get()), 0);
70  AES_ecb_encrypt(key_id_converted.data(), encrypted_key_id.data(),
71  aes_key.get(), AES_ENCRYPT);
72  std::string checksum =
73  std::string(encrypted_key_id.begin(), encrypted_key_id.end())
74  .substr(0, 8);
75  std::string base64_checksum;
76  base::Base64Encode(checksum, &base64_checksum);
77  std::string base64_key_id;
78  base::Base64Encode(
79  std::string(key_id_converted.begin(), key_id_converted.end()),
80  &base64_key_id);
81  std::string playready_header = kPlayHeaderObject_4_0;
82  base::ReplaceFirstSubstringAfterOffset(&playready_header, 0, "$0",
83  base64_key_id);
84  base::ReplaceFirstSubstringAfterOffset(&playready_header, 0, "$1",
85  base64_checksum);
86 
87  // Create a PlayReady Record.
88  // Outline in section '2.PlayReady Records' of
89  // 'PlayReady Header Object' document. Note data is in little endian format.
90  std::vector<uint16_t> record_value =
91  std::vector<uint16_t>(playready_header.begin(), playready_header.end());
92  shaka::media::BufferWriter writer_pr_record;
93  uint16_t record_type =
94  1; // Indicates that the record contains a rights management header.
95  uint16_t record_length = record_value.size() * 2;
96  writer_pr_record.AppendInt(static_cast<uint8_t>(record_type & 0xff));
97  writer_pr_record.AppendInt(static_cast<uint8_t>((record_type >> 8) & 0xff));
98  writer_pr_record.AppendInt(static_cast<uint8_t>(record_length & 0xff));
99  writer_pr_record.AppendInt(static_cast<uint8_t>((record_length >> 8) & 0xff));
100  for (auto record_item : record_value) {
101  writer_pr_record.AppendInt(static_cast<uint8_t>(record_item & 0xff));
102  writer_pr_record.AppendInt(static_cast<uint8_t>((record_item >> 8) & 0xff));
103  }
104 
105  // Create the PlayReady Header object.
106  // Outline in section '1.PlayReady Header Objects' of
107  // 'PlayReady Header Object' document. Note data is in little endian format.
108  shaka::media::BufferWriter writer_pr_header_object;
109  uint32_t playready_header_length = writer_pr_record.Size() + 4 + 2;
110  uint16_t record_count = 1;
111  writer_pr_header_object.AppendInt(
112  static_cast<uint8_t>(playready_header_length & 0xff));
113  writer_pr_header_object.AppendInt(
114  static_cast<uint8_t>((playready_header_length >> 8) & 0xff));
115  writer_pr_header_object.AppendInt(
116  static_cast<uint8_t>((playready_header_length >> 16) & 0xff));
117  writer_pr_header_object.AppendInt(
118  static_cast<uint8_t>((playready_header_length >> 24) & 0xff));
119  writer_pr_header_object.AppendInt(static_cast<uint8_t>(record_count & 0xff));
120  writer_pr_header_object.AppendInt(
121  static_cast<uint8_t>((record_count >> 8) & 0xff));
122  writer_pr_header_object.AppendBuffer(writer_pr_record);
123  *output = std::vector<uint8_t>(
124  writer_pr_header_object.Buffer(),
125  writer_pr_header_object.Buffer() + writer_pr_header_object.Size());
126  return Status::OK;
127 }
