165  // Find offset from start of data to next NALU start code
166  // and size of found start code (3 or 4 bytes).
167  // If no start code is found, offset is pointing to the first unprocessed byte
168  // (i.e. the first byte that was not considered as a possible start of a start
169  // code) and |*start_code_size| is set to 0.
170  // Postconditions:
171  // - |*offset| is between 0 and |data_size| included.
172  // It is strictly less than |data_size| if |data_size| > 0.
173  // - |*start_code_size| is either 0, 3 or 4.
174  static bool FindStartCode(const uint8_t* data,
175  uint64_t data_size,
176  uint64_t* offset,
177  uint8_t* start_code_size);
178 
184  Result Advance(Nalu* nalu);
185 
187  bool StartsWithStartCode();
188 
189  private:
190  enum Format {
191  kAnnexbByteStreamFormat,
192  kNalUnitStreamFormat
193  };
194 
195  // Move the stream pointer to the beginning of the next NALU,
196  // i.e. pointing at the next start code.
197  // Return true if a NALU has been found.
198  // If a NALU is found:
199  // - its size in bytes is returned in |*nalu_size| and includes
200  // the start code as well as the trailing zero bits.
201  // - the size in bytes of the start code is returned in |*start_code_size|.
202  bool LocateNaluByStartCode(uint64_t* nalu_size, uint8_t* start_code_size);
203 
204  // Pointer to the current NALU in the stream.
205  const uint8_t* stream_;
206  // The remaining size of the stream.
207  uint64_t stream_size_;
208  // The type of NALU being read.
