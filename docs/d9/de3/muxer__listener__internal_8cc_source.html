28 // This will return a positive value, given that |file_size| and
29 // |duration_seconds| are positive.
30 uint32_t EstimateRequiredBandwidth(uint64_t file_size, float duration_seconds) {
31  const uint64_t file_size_bits = file_size * 8;
32  const float bits_per_second = file_size_bits / duration_seconds;
33 
34  // Note that casting |bits_per_second| to an integer might make it 0. Take the
35  // ceiling and make sure that it returns a positive value.
36  return static_cast<uint32_t>(ceil(bits_per_second));
37 }

106  if (!language.empty() && language != "und") {

110  const std::vector<uint8_t>& codec_config = audio_stream_info->codec_config();
111  if (!codec_config.empty()) {
112  audio_info->set_decoder_config(&codec_config[0], codec_config.size());
113  }

116  uint32_t ec3_channel_map;

118  LOG(ERROR) << "Failed to calculate EC3 channel map.";

142 void SetMediaInfoStreamInfo(const StreamInfo& stream_info,
143  MediaInfo* media_info) {
144  if (stream_info.stream_type() == kStreamAudio) {
145  AddAudioInfo(static_cast<const AudioStreamInfo*>(&stream_info),
146  media_info);
147  } else if (stream_info.stream_type() == kStreamText) {
148  AddTextInfo(static_cast<const TextStreamInfo&>(stream_info), media_info);
149  } else {
150  DCHECK_EQ(stream_info.stream_type(), kStreamVideo);
151  AddVideoInfo(static_cast<const VideoStreamInfo*>(&stream_info),
152  media_info);
153  }
154  if (stream_info.duration() > 0) {
155  // |stream_info.duration()| contains the media duration from the original
156  // media header, which is usually good enough.
157  media_info->set_media_duration_seconds(
158  static_cast<double>(stream_info.duration()) / stream_info.time_scale());
159  }
160 }

193 bool SetVodInformation(const MuxerListener::MediaRanges& media_ranges,
194  float duration_seconds,
195  MediaInfo* media_info) {
196  DCHECK(media_info);
197 
198  if (duration_seconds <= 0.0f) {
199  // Non positive second media must be invalid media.
200  LOG(ERROR) << "Duration is not positive: " << duration_seconds;
201  return false;
202  }
203 
204 
205  if (media_ranges.init_range) {
206  SetRange(media_ranges.init_range->start, media_ranges.init_range->end,
207  media_info->mutable_init_range());
208  }
209 
210  if (media_ranges.index_range) {
211  SetRange(media_ranges.index_range->start, media_ranges.index_range->end,
212  media_info->mutable_index_range());
213  }
214 
215  media_info->set_media_duration_seconds(duration_seconds);
216 
217  if (!media_info->has_bandwidth()) {
218  // Calculate file size from media_ranges.
219  uint64_t file_size = 0;
220  if (media_ranges.init_range)
221  file_size = std::max(file_size, media_ranges.init_range->end + 1);
222  if (media_ranges.index_range)
223  file_size = std::max(file_size, media_ranges.index_range->end + 1);
224  if (!media_ranges.subsegment_ranges.empty()) {
225  file_size =
226  std::max(file_size, media_ranges.subsegment_ranges.back().end + 1);
227  }
228 
229  media_info->set_bandwidth(
230  EstimateRequiredBandwidth(file_size, duration_seconds));
231  }
232  return true;
233 }

235 void SetContentProtectionFields(
236  FourCC protection_scheme,
237  const std::vector<uint8_t>& default_key_id,

246  protected_content->set_default_key_id(default_key_id.data(),
247  default_key_id.size());
248  }
249 
250  for (const ProtectionSystemSpecificInfo& info : key_system_info) {
251  MediaInfo::ProtectedContent::ContentProtectionEntry* entry =
252  protected_content->add_content_protection_entry();
253  if (!info.system_id().empty())
254  entry->set_uuid(CreateUUIDString(info.system_id()));
255 
256  const std::vector<uint8_t> pssh = info.CreateBox();
257  entry->set_pssh(pssh.data(), pssh.size());
258  }

261 std::string CreateUUIDString(const std::vector<uint8_t>& data) {
262  DCHECK_EQ(16u, data.size());
263  std::string uuid =
264  base::ToLowerASCII(base::HexEncode(data.data(), data.size()));
265  uuid.insert(20, "-");
266  uuid.insert(16, "-");
267  uuid.insert(12, "-");
268  uuid.insert(8, "-");
269  return uuid;
