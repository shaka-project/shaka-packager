85 bool VideoStreamInfo::IsValidConfig() const {
86  return codec_ != kUnknownVideoCodec &&
87  width_ > 0 && width_ <= limits::kMaxDimension &&
88  height_ > 0 && height_ <= limits::kMaxDimension &&
