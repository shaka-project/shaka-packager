44 bool EsParserH264::ProcessNalu(const Nalu& nalu,
45  bool* is_key_frame,
46  int* pps_id_for_access_unit) {
47  switch (nalu.type()) {
48  case Nalu::H264_AUD: {
49  DVLOG(LOG_LEVEL_ES) << "Nalu: AUD";
50  break;
51  }
52  case Nalu::H264_SPS: {
53  DVLOG(LOG_LEVEL_ES) << "Nalu: SPS";
54  int sps_id;
55  if (h264_parser_->ParseSps(nalu, &sps_id) != H264Parser::kOk)
56  return false;
57  decoder_config_check_pending_ = true;
58  break;
59  }
60  case Nalu::H264_PPS: {
61  DVLOG(LOG_LEVEL_ES) << "Nalu: PPS";
62  int pps_id;
63  if (h264_parser_->ParsePps(nalu, &pps_id) != H264Parser::kOk) {
64  // Allow PPS parsing to fail if waiting for SPS.
65  if (last_video_decoder_config_)
66  return false;
67  } else {
68  decoder_config_check_pending_ = true;
69  }
70  break;
71  }
72  case Nalu::H264_IDRSlice:
73  case Nalu::H264_NonIDRSlice: {
74  *is_key_frame = (nalu.type() == Nalu::H264_IDRSlice);
75  DVLOG(LOG_LEVEL_ES) << "Nalu: slice IDR=" << is_key_frame;
76  H264SliceHeader shdr;
77  if (h264_parser_->ParseSliceHeader(nalu, &shdr) != H264Parser::kOk) {
78  // Only accept an invalid SPS/PPS at the beginning when the stream
79  // does not necessarily start with an SPS/PPS/IDR.
80  if (last_video_decoder_config_)
81  return false;
82  } else {
83  *pps_id_for_access_unit = shdr.pic_parameter_set_id;
84  }
85  break;
86  }
87  default: {
88  DVLOG(LOG_LEVEL_ES) << "Nalu: " << nalu.type();
89  }
90  }
91 
92  return true;
93 }

95 bool EsParserH264::UpdateVideoDecoderConfig(int pps_id) {
96  // Update the video decoder configuration if needed.
97  if (!decoder_config_check_pending_)
98  return true;
99 
100  const H264Pps* pps = h264_parser_->GetPps(pps_id);
101  const H264Sps* sps;
102  if (!pps) {
103  // Only accept an invalid PPS at the beginning when the stream
104  // does not necessarily start with an SPS/PPS/IDR.
105  // In this case, the initial frames are conveyed to the upper layer with
106  // an invalid VideoDecoderConfig and it's up to the upper layer
107  // to process this kind of frame accordingly.
108  return last_video_decoder_config_ == nullptr;
109  } else {
110  sps = h264_parser_->GetSps(pps->seq_parameter_set_id);
111  if (!sps)
112  return false;
113  decoder_config_check_pending_ = false;
114  }
115 
116  std::vector<uint8_t> decoder_config_record;
117  if (!stream_converter()->GetDecoderConfigurationRecord(
118  &decoder_config_record)) {
119  DLOG(ERROR) << "Failure to construct an AVCDecoderConfigurationRecord";
120  return false;
121  }
122 
123  if (last_video_decoder_config_) {
124  if (last_video_decoder_config_->extra_data() != decoder_config_record) {
125  // Video configuration has changed. Issue warning.
126  // TODO(tinskip): Check the nature of the configuration change. Only
127  // minor configuration changes (such as frame ordering) can be handled
128  // gracefully by decoders without notification. Major changes (such as
129  // video resolution changes) should be treated as errors.
130  LOG(WARNING) << "H.264 decoder configuration has changed.";
131  last_video_decoder_config_->set_extra_data(decoder_config_record);
132  }
133  return true;
134  }
135 
136  uint32_t coded_width = 0;
137  uint32_t coded_height = 0;
138  uint32_t pixel_width = 0;
139  uint32_t pixel_height = 0;
140  if (!ExtractResolutionFromSps(*sps, &coded_width, &coded_height, &pixel_width,
141  &pixel_height)) {
142  LOG(ERROR) << "Failed to parse SPS.";
143  return false;
144  }
145 
146  last_video_decoder_config_ = scoped_refptr<StreamInfo>(
147  new VideoStreamInfo(
148  pid(),
149  kMpeg2Timescale,
150  kInfiniteDuration,
151  kCodecH264,
152  AVCDecoderConfiguration::GetCodecString(decoder_config_record[1],
153  decoder_config_record[2],
154  decoder_config_record[3]),
155  std::string(),
156  coded_width,
157  coded_height,
158  pixel_width,
159  pixel_height,
160  0,
161  H264ByteToUnitStreamConverter::kUnitStreamNaluLengthSize,
162  decoder_config_record.data(),
163  decoder_config_record.size(),
164  false));
165  DVLOG(1) << "Profile IDC: " << sps->profile_idc;
166  DVLOG(1) << "Level IDC: " << sps->level_idc;
167  DVLOG(1) << "log2_max_frame_num_minus4: " << sps->log2_max_frame_num_minus4;
168 
169  // Video config notification.
170  new_stream_info_cb_.Run(last_video_decoder_config_);
171 
172  return true;
173 }
