68 bool DashIopMpdNotifier::NotifyNewContainer(const MediaInfo& media_info,
69  uint32_t* container_id) {
70  DCHECK(container_id);
71 
72  ContentType content_type = GetContentType(media_info);
73  if (content_type == kContentTypeUnknown)
74  return false;
75 
76  base::AutoLock auto_lock(lock_);
77  std::string lang;
78  if (media_info.has_audio_info()) {
79  lang = media_info.audio_info().language();
80  } else if (media_info.has_text_info()) {
81  lang = media_info.text_info().language();
82  }
83 
84  AdaptationSet* adaptation_set =
85  GetAdaptationSetForMediaInfo(media_info, content_type, lang);
86  DCHECK(adaptation_set);
87  if (media_info.has_text_info()) {
88  // IOP requires all AdaptationSets to have (sub)segmentAlignment set to
89  // true, so carelessly set it to true.
90  // In practice it doesn't really make sense to adapt between text tracks.
91  adaptation_set->ForceSetSegmentAlignment(true);
92  }
93 
94  MediaInfo adjusted_media_info(media_info);
95  MpdBuilder::MakePathsRelativeToMpd(output_path_, &adjusted_media_info);
96  Representation* representation =
97  adaptation_set->AddRepresentation(adjusted_media_info);
98  if (!representation)
99  return false;
100 
101  representation_id_to_adaptation_set_[representation->id()] = adaptation_set;
102 
103  SetGroupId(content_type, lang, adaptation_set);
104 
105  *container_id = representation->id();
106  DCHECK(!ContainsKey(representation_map_, representation->id()));
107  representation_map_[representation->id()] = representation;
108  return true;
109 }

171 AdaptationSet* DashIopMpdNotifier::GetAdaptationSetForMediaInfo(
172  const MediaInfo& media_info,
173  ContentType content_type,
174  const std::string& language) {
175  std::list<AdaptationSet*>& adaptation_sets =
176  adaptation_set_list_map_[content_type][language];
177  if (adaptation_sets.empty())
178  return NewAdaptationSet(media_info, language, &adaptation_sets);
179 
180  const bool has_protected_content = media_info.has_protected_content();
181 
182  for (std::list<AdaptationSet*>::const_iterator adaptation_set_it =
183  adaptation_sets.begin();
184  adaptation_set_it != adaptation_sets.end(); ++adaptation_set_it) {
185  ProtectedContentMap::const_iterator protected_content_it =
186  protected_content_map_.find((*adaptation_set_it)->id());
187 
188  // If the AdaptationSet ID is not registered in the map, then it is clear
189  // content (or encrypted but doesn't need <ContentProtection> element
190  // possibly because the player knows how to handle it).
191  if (protected_content_it == protected_content_map_.end()) {
192  // Can reuse the AdaptationSet without content protection.
193  if (!has_protected_content)
194  return *adaptation_set_it;
195  continue;
196  }
197 
198  if (ProtectedContentEq(protected_content_it->second,
199  media_info.protected_content())) {
200  // Content protection info matches. Reuse the AdaptationSet.
201  return *adaptation_set_it;
202  }
203  }
204 
205  // None of the adaptation sets match with the new content protection.
206  // Need a new one.
207  return NewAdaptationSet(media_info, language, &adaptation_sets);
208 }

210 // Get all the UUIDs of the AdaptationSet. If another AdaptationSet has the
211 // same UUIDs then those should be groupable.
212 void DashIopMpdNotifier::SetGroupId(ContentType type,
213  const std::string& language,
214  AdaptationSet* adaptation_set) {
215  if (adaptation_set->Group() >= 0) // @group already assigned.
216  return;
217 
218  ProtectedContentMap::const_iterator protected_content_it =
219  protected_content_map_.find(adaptation_set->id());
220  // Clear contents should be in one AdaptationSet, so no need to assign
221  // @group.
222  if (protected_content_it == protected_content_map_.end()) {
223  DVLOG(1) << "No content protection set for AdaptationSet@id="
224  << adaptation_set->id();
225  return;
226  }
227 
228  // Get all the UUIDs of the ContentProtections in AdaptationSet.
229  std::set<std::string> adaptation_set_uuids =
230  GetUUIDs(protected_content_it->second);
231 
232  std::list<AdaptationSet*>& same_type_adapatation_sets =
233  adaptation_set_list_map_[type][language];
234  DCHECK(!same_type_adapatation_sets.empty())
235  << "same_type_adapatation_sets should not be null, it should at least "
236  "contain adaptation_set";
237 
238  for (std::list<AdaptationSet*>::iterator adaptation_set_it =
239  same_type_adapatation_sets.begin();
240  adaptation_set_it != same_type_adapatation_sets.end();
241  ++adaptation_set_it) {
242  const uint32_t loop_adaptation_set_id = (*adaptation_set_it)->id();
243  if (loop_adaptation_set_id == adaptation_set->id() ||
244  !ContainsKey(protected_content_map_, loop_adaptation_set_id)) {
245  continue;
246  }
247 
248  const MediaInfo::ProtectedContent& loop_protected_content =
249  protected_content_map_[loop_adaptation_set_id];
250  if (static_cast<int>(adaptation_set_uuids.size()) !=
251  loop_protected_content.content_protection_entry().size()) {
252  // Different number of UUIDs, cannot be grouped.
253  continue;
254  }
255 
256  if (adaptation_set_uuids == GetUUIDs(loop_protected_content)) {
257  AdaptationSet& uuid_match_adaptation_set = **adaptation_set_it;
258  // They match. These AdaptationSets can be in the same group. Break out.
259  if (uuid_match_adaptation_set.Group() >= 0) {
260  adaptation_set->SetGroup(uuid_match_adaptation_set.Group());
261  } else {
262  const int group_id = next_group_id_++;
263  uuid_match_adaptation_set.SetGroup(group_id);
264  adaptation_set->SetGroup(group_id);
265  }
266  break;
267  }
268  }

271 AdaptationSet* DashIopMpdNotifier::NewAdaptationSet(
272  const MediaInfo& media_info,
273  const std::string& language,
274  std::list<AdaptationSet*>* adaptation_sets) {
275  AdaptationSet* new_adaptation_set = mpd_builder_->AddAdaptationSet(language);
276  if (media_info.has_protected_content()) {
277  DCHECK(!ContainsKey(protected_content_map_, new_adaptation_set->id()));
278  protected_content_map_[new_adaptation_set->id()] =
279  media_info.protected_content();
280  AddContentProtectionElements(media_info, new_adaptation_set);
281  }
282  adaptation_sets->push_back(new_adaptation_set);
283 
284  if (media_info.has_video_info()) {
285  // Because 'lang' is ignored for videos, |adaptation_sets| must have
286  // all the video AdaptationSets.
287  if (adaptation_sets->size() > 2) {
288  new_adaptation_set->AddRole(AdaptationSet::kRoleMain);
289  } else if (adaptation_sets->size() == 2) {
290  // Set "main" Role for both AdaptatoinSets.
291  (*adaptation_sets->begin())->AddRole(AdaptationSet::kRoleMain);
292  new_adaptation_set->AddRole(AdaptationSet::kRoleMain);
293  }
294  }
295  return new_adaptation_set;
296 }
