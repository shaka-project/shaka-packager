33 // These methods work together to notify that the media is encrypted.
34 // If OnEncryptionInfoReady() is called before the media has been started, then
35 // the information is stored and handled when OnEncryptionStart() is called.
36 // if OnEncryptionStart() is called before the media has been started then
37 // OnMediaStart() is responsible for notifying that the segments are encrypted
38 // right away i.e. call OnEncryptionStart().
39 // For now (because Live HLS is not implemented yet) this should be called once,
40 // before media is started. So the logic after the first if statement should not
41 // be taken.
42 void HlsNotifyMuxerListener::OnEncryptionInfoReady(
43  bool is_initial_encryption_info,
44  FourCC protection_scheme,
45  const std::vector<uint8_t>& key_id,
46  const std::vector<uint8_t>& iv,
47  const std::vector<ProtectionSystemSpecificInfo>& key_system_infos) {
48  if (!media_started_) {
49  next_key_id_ = key_id;
50  next_iv_ = iv;
51  next_key_system_infos_ = key_system_infos;
52  protection_scheme_ = protection_scheme;
53  return;
54  }
55  for (const ProtectionSystemSpecificInfo& info : key_system_infos) {
56  const bool result = hls_notifier_->NotifyEncryptionUpdate(
57  stream_id_, key_id, info.system_id(), iv, info.CreateBox());
58  LOG_IF(WARNING, !result) << "Failed to add encryption info.";
59  }
60 }

62 void HlsNotifyMuxerListener::OnEncryptionStart() {
63  if (!media_started_) {
64  must_notify_encryption_start_ = true;
65  return;
66  }
67  if (next_key_id_.empty()) {
68  DCHECK(next_iv_.empty());
69  DCHECK(next_key_system_infos_.empty());
70  return;
71  }
72 
73  for (const ProtectionSystemSpecificInfo& info : next_key_system_infos_) {
74  const bool result = hls_notifier_->NotifyEncryptionUpdate(
75  stream_id_, next_key_id_, info.system_id(), next_iv_,
76  info.CreateBox());
77  LOG_IF(WARNING, !result) << "Failed to add encryption info";
78  }
79  next_key_id_.clear();
80  next_iv_.clear();
81  next_key_system_infos_.clear();
82  must_notify_encryption_start_ = false;
83 }

85 void HlsNotifyMuxerListener::OnMediaStart(const MuxerOptions& muxer_options,
86  const StreamInfo& stream_info,
87  uint32_t time_scale,
88  ContainerType container_type) {
89  MediaInfo media_info;
90  if (!internal::GenerateMediaInfo(muxer_options, stream_info, time_scale,
91  container_type, &media_info)) {
92  LOG(ERROR) << "Failed to generate MediaInfo from input.";
93  return;
94  }
95  if (protection_scheme_ != FOURCC_NULL) {
96  internal::SetContentProtectionFields(protection_scheme_, next_key_id_,
97  next_key_system_infos_, &media_info);
98  }
99 
100  const bool result = hls_notifier_->NotifyNewStream(
101  media_info, playlist_name_, ext_x_media_name_, ext_x_media_group_id_,
102  &stream_id_);
103  if (!result) {
104  LOG(WARNING) << "Failed to notify new stream.";
105  return;
106  }
107 
108  media_started_ = true;
109  if (must_notify_encryption_start_) {
110  OnEncryptionStart();
111  }
112 }
