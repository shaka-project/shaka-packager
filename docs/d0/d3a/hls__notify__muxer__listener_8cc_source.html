33 // These methods work together to notify that the media is encrypted.
34 // If OnEncryptionInfoReady() is called before the media has been started, then
35 // the information is stored and handled when OnEncryptionStart() is called.
36 // If OnEncryptionStart() is called before the media has been started then
37 // OnMediaStart() is responsible for notifying that the segments are encrypted
38 // right away i.e. call OnEncryptionStart().
39 void HlsNotifyMuxerListener::OnEncryptionInfoReady(
40  bool is_initial_encryption_info,
41  FourCC protection_scheme,
42  const std::vector<uint8_t>& key_id,
43  const std::vector<uint8_t>& iv,
44  const std::vector<ProtectionSystemSpecificInfo>& key_system_infos) {
45  if (!media_started_) {
46  next_key_id_ = key_id;
47  next_iv_ = iv;
48  next_key_system_infos_ = key_system_infos;
49  protection_scheme_ = protection_scheme;
50  return;
51  }
52  for (const ProtectionSystemSpecificInfo& info : key_system_infos) {
53  const bool result = hls_notifier_->NotifyEncryptionUpdate(
54  stream_id_, key_id, info.system_id(), iv, info.CreateBox());
55  LOG_IF(WARNING, !result) << "Failed to add encryption info.";
56  }
57 }

59 void HlsNotifyMuxerListener::OnEncryptionStart() {
60  if (!media_started_) {
61  must_notify_encryption_start_ = true;
62  return;
63  }
64  if (next_key_id_.empty()) {
65  DCHECK(next_iv_.empty());
66  DCHECK(next_key_system_infos_.empty());
67  return;
68  }
69 
70  for (const ProtectionSystemSpecificInfo& info : next_key_system_infos_) {
71  const bool result = hls_notifier_->NotifyEncryptionUpdate(
72  stream_id_, next_key_id_, info.system_id(), next_iv_,
73  info.CreateBox());
74  LOG_IF(WARNING, !result) << "Failed to add encryption info";
75  }
76  next_key_id_.clear();
77  next_iv_.clear();
78  next_key_system_infos_.clear();
79  must_notify_encryption_start_ = false;
80 }

82 void HlsNotifyMuxerListener::OnMediaStart(const MuxerOptions& muxer_options,
83  const StreamInfo& stream_info,
84  uint32_t time_scale,
85  ContainerType container_type) {
86  MediaInfo media_info;
87  if (!internal::GenerateMediaInfo(muxer_options, stream_info, time_scale,
88  container_type, &media_info)) {
89  LOG(ERROR) << "Failed to generate MediaInfo from input.";
90  return;
91  }
92  if (protection_scheme_ != FOURCC_NULL) {
93  internal::SetContentProtectionFields(protection_scheme_, next_key_id_,
94  next_key_system_infos_, &media_info);
95  }
96 
97  media_info_ = media_info;
98  if (!media_info_.has_segment_template()) {
99  return;
100  }
101 
102  const bool result = hls_notifier_->NotifyNewStream(
103  media_info, playlist_name_, ext_x_media_name_, ext_x_media_group_id_,
104  &stream_id_);
105  if (!result) {
106  LOG(WARNING) << "Failed to notify new stream.";
107  return;
108  }
109 
110  media_started_ = true;
111  if (must_notify_encryption_start_) {
112  OnEncryptionStart();
113  }
114 }

118 void HlsNotifyMuxerListener::OnMediaEnd(const MediaRanges& media_ranges,
119  float duration_seconds) {
120  // TODO(kqyang): Should we just Flush here to avoid calling Flush explicitly?
121  // Don't flush the notifier here. Flushing here would write all the playlists
122  // before all Media Playlists are read. Which could cause problems
123  // setting the correct EXT-X-TARGETDURATION.
124  if (media_info_.has_segment_template()) {
125  return;
126  }
127  if (media_ranges.init_range) {
128  shaka::Range* init_range = media_info_.mutable_init_range();
129  init_range->set_begin(media_ranges.init_range.value().start);
130  init_range->set_end(media_ranges.init_range.value().end);
131  }
132  if (media_ranges.index_range) {
133  shaka::Range* index_range = media_info_.mutable_index_range();
134  index_range->set_begin(media_ranges.index_range.value().start);
135  index_range->set_end(media_ranges.index_range.value().end);
136  }
137 
138  // TODO(rkuroiwa): Make this a method. This is the same as OnMediaStart().
139  const bool result = hls_notifier_->NotifyNewStream(
140  media_info_, playlist_name_, ext_x_media_name_, ext_x_media_group_id_,
141  &stream_id_);
142  if (!result) {
143  LOG(WARNING) << "Failed to notify new stream for VOD.";
144  return;
145  }
146 
147  // TODO(rkuroiwa); Keep track of which (sub)segments are encrypted so that the
148  // notification is sent right before the enecrypted (sub)segments.
149  media_started_ = true;
150  if (must_notify_encryption_start_) {
151  OnEncryptionStart();
152  }
153 
154  if (!media_ranges.subsegment_ranges.empty()) {
155  const std::vector<Range>& subsegment_ranges =
156  media_ranges.subsegment_ranges;
157  size_t num_subsegments = subsegment_ranges.size();
158  if (segment_infos_.size() != num_subsegments) {
159  LOG(WARNING) << "Number of subsegment ranges (" << num_subsegments
160  << ") does not match the number of subsegments notified to "
161  "OnNewSegment() ("
162  << segment_infos_.size() << ").";
163  num_subsegments = std::min(segment_infos_.size(), num_subsegments);
164  }
165  for (size_t i = 0; i < num_subsegments; ++i) {
166  const Range& range = subsegment_ranges[i];
167  const SegmentInfo& subsegment_info = segment_infos_[i];
168  hls_notifier_->NotifyNewSegment(
169  stream_id_, media_info_.media_file_name(), subsegment_info.start_time,
170  subsegment_info.duration, range.start, range.end + 1 - range.start);
171  }
172  }
173 }

191  LOG_IF(WARNING, !result) << "Failed to add new segment.";
192 }
