112 const inline std::string FourCCToString(FourCC fourcc) {
113  char buf[5];
114  buf[0] = (fourcc >> 24) & 0xff;
115  buf[1] = (fourcc >> 16) & 0xff;
116  buf[2] = (fourcc >> 8) & 0xff;
117  buf[3] = (fourcc) & 0xff;
118  buf[4] = 0;
119  return std::string(buf);
120 }
