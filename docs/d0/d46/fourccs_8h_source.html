150 const inline std::string FourCCToString(FourCC fourcc) {
151  char buf[5];
152  buf[0] = (fourcc >> 24) & 0xff;
153  buf[1] = (fourcc >> 16) & 0xff;
154  buf[2] = (fourcc >> 8) & 0xff;
155  buf[3] = (fourcc) & 0xff;
156  buf[4] = 0;
157  return std::string(buf);
158 }
